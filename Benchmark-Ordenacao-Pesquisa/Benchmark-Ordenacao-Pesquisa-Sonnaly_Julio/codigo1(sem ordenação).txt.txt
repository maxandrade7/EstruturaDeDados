# Adicionar as funções de Ordenamento e Pesquisa.
def bubbleSort(lista):
    comparacoes = 0
    trocas = 0
    for i in range(len(lista)):
        for c in range(i+1,len(lista)):
            comparacoes+=1
            if(lista[c] < lista[i]):
                trocas+=1
                aux = lista[i]
                lista[i] = lista[c]
                lista[c] = aux
    print("COMP: %d"%(comparacoes))
    print("TROCAS: %d"%(trocas))

    return lista
